#include <cstdio>
#include <cstring>
#include <cassert>
#include <algorithm>
#include <set>
#include <map>
#include <vector>
using namespace std;

const int MAXN = 30000000 + 5;

struct DestSeg{
    int st, ed;
    DestSeg *nxt;
};
struct Node{
    DestSeg *head;
    int deg_o;
};

int id[MAXN];
Node nodes[MAXN]; int n;
Node *getnode(int x){
    if (!id[x]){
        id[x] = ++n;
        assert(n <= MAXN);
    }
    return &nodes[id[x]];
}

vector<int> dest; int m;
void add_edge(int u, int v){
    Node *u_node = getnode(u), *v_node = getnode(v);
    if (u_node->head && u_node->head->ed == m){
        dest.push_back(id[v]); m++;
        u_node->head->ed++;
    }
    else {
        DestSeg *newseg = new DestSeg;
        newseg->st = m++;
        dest.push_back(id[v]);
        newseg->ed = m;
        newseg->nxt = u_node->head;
        u_node->head = newseg;
    }
    u_node->deg_o++;
}

FILE *fp;
const int BUFSIZE = 1<<16;
char my_getchar(){
    static char buf[BUFSIZE+5];
    static int pos = 0, size = 0;
    if (pos == size){
        pos = 0;
        size = fread(buf, 1, BUFSIZE, fp);
        if (size == 0) return EOF;
    }
    return buf[pos++];
}
void load_graph(){
    fp = stdin;
    while (1){
        char ch;
        int x = 0, head = -1;
        do{
            ch = my_getchar();
            if (ch >= '0' && ch <= '9')
                x = x * 10 + ch - '0';
            else {
                if (x == 0)
                    continue;
                if (head == -1){
                    head = x;
                    getnode(head);
                } else
                    add_edge(head, x);
                x = 0;
            }
        } while (ch != '\n' && ch != EOF);
        if (ch == EOF)
            break;
    }
}

double pagerank[MAXN], pagerank_[MAXN];
void calc_pagerank(double alpha = 0.15, double num_loop = 40){
    double tmp = 1.0 / n;
    for (int i = 1; i <= n; ++i)
        pagerank[i] = tmp;

    for (int i = 0; i < num_loop; ++i){
        double sum_scatter = 0;
        memset(pagerank_, 0, sizeof(pagerank_));
        for (int u = 1; u <= n; ++u)
            if (nodes[u].deg_o){
                double tmp = (1 - alpha) * pagerank[u] / nodes[u].deg_o;
                for (DestSeg *seg = nodes[u].head; seg; seg = seg->nxt)
                    for (int i = seg->st; i < seg->ed; ++i)
                        pagerank_[dest[i]] += tmp;
            } else
                sum_scatter += pagerank[u];
        sum_scatter = (alpha + (1 - alpha) * sum_scatter) / n;
        for (int i = 1; i <= n; ++i)
            pagerank[i] = pagerank_[i] + sum_scatter;
    }       
}

void output(){
    for (int i = 1; i < MAXN; ++i)
        if (id[i])
            printf("%d %.10lf\n", i, pagerank[id[i]]);
}

int main(){
    load_graph();
    calc_pagerank(0.15, 100);
    output();
    return 0;
}
